ALTER FUNCTION [dbo].[fnGetDurationCost](@StartTime datetime,@EndTime datetime,@BreakMinutes int,@OverTimeMinutes int,@StdRateSalary decimal(10,4),@OverTimeRate decimal(10,4))
returns decimal (10,4)
AS   
BEGIN  
    DECLARE @RetVal decimal (10,4);  
	DECLARE @Duration int;
	DECLARE @OverTimeHr decimal (10,4);
	DECLARE @NetDuration decimal (10,4);
	DECLARE @DurationHr decimal (10,4);
	Declare @BreakMinutesDecimal decimal (10,4);

	if (@StartTime IS NOT NULL AND @EndTime IS NOT NULL)
	
	BEGIN
	 SET @Duration = DATEDIFF(MI,@StartTime,@EndTime);
	 set @BreakMinutesDecimal = cast (@BreakMinutes as decimal(10,4))
	 SET @DurationHr = Cast( ((@Duration - @BreakMinutesDecimal)/Cast(60 as decimal (10,4))) as decimal(10,4));
     SET @OverTimeHr = Cast( (@OverTimeMinutes / Cast(60 as decimal(10,4))) as decimal(10,4));
	 
	 set @NetDuration=@DurationHr
	 if @DurationHr>=@OverTimeHr

	 	 SET @NetDuration=cast ( (@DurationHr - @OverTimeHr) as decimal(10,4));
	 else
	
		 set @NetDuration=@DurationHr



	 set @RetVal = (@NetDuration * @StdRateSalary) + (@OverTimeHr * @OverTimeRate);

	 END;

    RETURN @RetVal;  
END; 
GO



ALTER view [dbo].[vwSalesOrderPhaseItemTimeSheetCostByOTSummary]
as
select
SOPI.SalesOrderPhaseItemID,
SUM(ISNULL(dbo.fnGetDurationCost(TSE.StartTime,TSE.EndTime,TSE.BreakMins,TSE.OverTimeMinutes,EROP.StandardRate,EROP.OverTimeRate),0)) TotalCost
from TimeSheetEntry TSE
INNER JOIN WorkOrder WO ON WO.WorkOrderID=TSE.WorkOrderID

INNER JOIN Employee EMP ON EMP.EmployeeID=TSE.EmployeeID
LEFT join EmployeeRateOfPay EROP on EROP.EmployeeID=EMP.EmployeeID

GROUP BY SOPI.SalesOrderPhaseItemID
GO
