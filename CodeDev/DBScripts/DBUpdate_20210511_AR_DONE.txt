WPI.Width,
WPI.Length,
WPI.Quantity,
WPI.QuantityUsed,
WPI.OutstandingQty,
WPI.VolumeM3,
wpi.Thickness,
WPI.StockItemID,

WPI.Description,


WP.WoodPalletID,
WP.PalletRef,
WP.CreatedDate,
WP.LocationID,
WP.RefPalletOutside,
WP.KilnStartDate,
WP.KilnEndDate,
